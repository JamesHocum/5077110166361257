using RestSharp;
using System.Threading.Tasks;

public interface IMyApiClient
{
    Task<string> ExecuteCommandAsync(string apiKey, string command);
}

// 4. Implement the API client:
public class MyApiClient : IMyApiClient
{
    private readonly string _baseUrl = "https://your-api-base-url.com"; // Replace

    public async Task<string> ExecuteCommandAsync(string apiKey, string command)
    {
        var client = new RestClient(_baseUrl);
        var request = new RestRequest("/your-api-endpoint", Method.Post); // Replace

        request.AddHeader("Authorization", $"Bearer {apiKey}");
        request.AddJsonBody(new { command = command }); // Assuming the API expects JSON

        try {
            RestResponse response = await client.ExecutePostAsync(request);
            if (response.IsSuccessful) {
                // Assuming the API returns a simple result string
                return response.Content;  // Or deserialize the JSON response
            } else {
                return $"API Error: {response.StatusCode} - {response.Content}";
            }
        } catch (Exception ex) {
            return $"Network error: {ex.Message}";
        }
    }
}

using Microsoft.VisualStudio.Shell;
using Microsoft.VisualStudio.Shell.Interop;
using System;
using System.ComponentModel.Design;
using System.Threading.Tasks;

public class MyCommand
{
    private readonly AsyncPackage package;
    private readonly IMyApiClient _apiClient = new MyApiClient();  // Instantiate API client

    private MyCommand(AsyncPackage package, OleMenuCommandService commandService)
    {
        this.package = package ?? throw new ArgumentNullException(nameof(package));
        commandService = commandService ?? throw new ArgumentNullException(nameof(commandService));

        var menuCommandID = new CommandID(CommandSet, CommandId);
        var menuItem = new MenuCommand(this.Execute, menuCommandID);
        commandService.AddCommand(menuItem);
    }

    public static MyCommand Instance {
        get;
        private set;
    }

    private static async Task InitializeAsync(AsyncPackage package)
    {
        // Verify the current thread is the UI thread - do not call this on a background thread
        ThreadHelper.ThrowIfNotOnUIThread(nameof(InitializeAsync));

        OleMenuCommandService commandService = await package.GetServiceAsync((typeof(IMenuCommandService))) as OleMenuCommandService;
        Instance = new MyCommand(package, commandService);
    }

    private void Execute(object sender, EventArgs e)
    {
        ThreadHelper.ThrowIfNotOnUIThread(nameof(Execute));
        // Get API key and command from the user (e.g., through a dialog box)
        string apiKey = "YOUR_API_KEY"; // Replace with user input or settings
        string command = "your_command";  // Replace with user input

        Task.Run(async () =>
        {
            string result = await _apiClient.ExecuteCommandAsync(apiKey, command);
            // Display the result in Visual Studio (e.g., in an output window or message box)
            ShowResultInOutputWindow(result);
        });
    }

    private void ShowResultInOutputWindow(string result)
    {
        ThreadHelper.ThrowIfNotOnUIThread(nameof(ShowResultInOutputWindow));

        IVsOutputWindow outputWindow = (IVsOutputWindow)package.GetService(typeof(SVsOutputWindow));
        if (outputWindow == null) return;

        Guid customGuid = new Guid("YOUR_OUTPUT_WINDOW_GUID"); // Replace with a unique GUID
        IVsOutputWindowPane pane;
        int hr = outputWindow.GetPane(ref customGuid, out pane);

        if (hr != 0 || pane == null)
        {
            // Create a new output window pane if it doesn't exist
            hr = outputWindow.CreatePane(ref customGuid, "My Extension Output", 1, 1);
            if (hr != 0) return;
            hr = outputWindow.GetPane(ref customGuid, out pane);
            if (hr != 0 || pane == null) return;
        }

        pane.Activate();
        pane.OutputStringThreadSafe(result + Environment.NewLine);
    }

    // Command Menu and Id (Replace with your own)
    public const int CommandId = 0x0100;
    public static readonly Guid CommandSet = new Guid("YOUR_COMMAND_SET_GUID"); // Replace
}
