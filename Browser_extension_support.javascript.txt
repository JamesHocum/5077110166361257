// Browser Extension (Popup.js - Example)
document.getElementById('executeButton').addEventListener('click', () => {
  const command = document.getElementById('commandInput').value;
  executeCommand(command);
});

function executeCommand(command) {
  //  Implement the command execution logic (using the browser's scripting capabilities)
  //  - Send a message to a content script
  //  - Receive the result and display it in the #output div
  console.log(`Executing command: ${command}`);
  // (Example - send message to content script)
  chrome.scripting.executeScript({
    target: { tabId: chrome.tabs.getCurrent().id },
    function: (command) => {
      // This code runs in the content script
      try {
        // Use fetch to execute the command (e.g., call an API, etc.)
        // Example:
        fetch("https://api.example.com/command", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ command: command })
        })
        .then(response => response.json())
        .then(data => {
          console.log(data);
          // Display the result in the browser's console or return it to the popup
          return data.result;
        })
        .catch(error => {
          console.error("Error executing command:", error);
          return "Error: " + error.message;
        });
      } catch (error) {
        console.error("Error executing command:", error);
        return "Error: " + error.message;
      }
    },
    args: [command]
  }, (results) => {
    if (chrome.runtime.lastError) {
      console.error(chrome.runtime.lastError);
      displayOutput("Error: " + chrome.runtime.lastError.message);
    } else {
      //results.forEach(result => { //results is an array, you may want to process it
        displayOutput(results[0].result);
      //});
    }
  });
}

function displayOutput(text) {
  const outputDiv = document.getElementById('output');
  const outputLine = document.createElement('div');
  outputLine.textContent = text; // or text.replace(/\n/g, '<br>') for line breaks
  outputDiv.appendChild(outputLine);
  outputDiv.scrollTop = outputDiv.scrollHeight; // Auto-scroll to bottom
}
